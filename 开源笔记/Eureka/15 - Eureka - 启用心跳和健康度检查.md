# 应用的名称
spring.application.name=eureka-client

server.port=30000

# 指定Eureka注册中心的地址
eureka.client.serviceUrl.defaultZone=http://localhost:20000/eureka

# 每隔5秒钟，向服务中心发送一条续约指令
# lease-renewal 续约更新的意思
# interval 间隔的意思
eureka.instance.lease-renewal-interval-in-seconds=5

# 如果30秒内，依然没有收到续约请求，判定服务过期（上西天）如果没开启服务自保
# 就把该服务剔除掉
eureka.instance.lease-expiration-duration-in-seconds=30

```text
spring.application.name=eureka-server

server.port=20000

eureka.instance.hostname=localhost
# 是否发起服务注册 我是注册中心 不用注册
eureka.client.register-with-eureka=false
# 是否去拉取服务注册表 作为注册中心 没有必要
eureka.client.fetch-registry=false

# 强制关闭服务自保机制 默认为 true（自动开关不起作用）
# self-preservation 自保的意思
eureka.server.enable-self-preservation=false

# 每隔多久触发一次剔除 不可用服务 没有心跳的服务 相当于一个job任务
# ms 是毫秒 eviction是剔除的意思 这里是10秒
eureka.server.eviction-interval-timer-in-ms=10000
```
